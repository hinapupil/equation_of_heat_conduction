\begin{align*} &u:温度, x:位置, t:時間, \\ &k:微小時間, h:微小距離, \\ &j:時間方向の変数, i:位置方向の変数 \\ \; \\ \frac{\partial u}{\partial t}&=\frac{\partial^2}{\partial x^2}\\ &\downarrow 差分近似 \\ \frac{u_{i,j+1}-u_{i,j}}{k}&=\frac{u_{i+1,j}-2u_{i,j}+u_{i-1,j}}{h^2} \cdot\cdot\cdot\ast \\ ここで, u_{i,0}&=T \; \; (i=0,1,...,i_{max}) \\ &とし、差分近似式 \ast を変形 \\ u_{i,j+1}&=r(u_{i+1,j}+u_{i-1,j})+(1-2r)u_{i,j} \\ & ただし, r=\frac{k}{h^2} \end{align*}
